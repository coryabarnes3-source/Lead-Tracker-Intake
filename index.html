Connect to Google Sheets
Takes about 2 minutes. You only do this once.
1
Open your Google Sheet
Go to the Lead Tracker sheet in Google Sheets on your computer.
2
Open Apps Script
Click Extensions → Apps Script in the menu bar.
3
Paste this script
Delete everything in the editor and paste the code below. Then click 💾 Save.
function doPost(e) { var sheet = SpreadsheetApp .getActiveSpreadsheet() .getSheetByName("Lead Tracker"); var data = JSON.parse(e.postData.contents); var row = [ data.rowNum, data.clientName, data.sourceCode, data.partnerName, data.clientPhone, data.clientEmail, data.notes, data.timestamp ]; sheet.appendRow(row); return ContentService .createTextOutput(JSON.stringify({"status":"ok"})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService .createTextOutput("Lead Tracker active"); }
